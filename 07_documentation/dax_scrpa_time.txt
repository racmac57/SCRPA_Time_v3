2. 7-Day Count Dynamic =
VAR SelectedCrime = SELECTEDVALUE('Crime Types Slicer'[Display Name])
VAR CrimeType = IF(ISBLANK(SelectedCrime), "All Crimes", SelectedCrime)

// Dynamic 7-day period based on today
VAR Today = TODAY()
VAR Current7DayStart = Today - 6  // Last 7 days
VAR Current7DayEnd = Today
VAR StartDateText = FORMAT(Current7DayStart, "MM/DD/YY")
VAR EndDateText = FORMAT(Current7DayEnd, "MM/DD/YY")

// Count ONLY incidents in dynamic 7-day period
VAR Count7Day =
    SWITCH(
        SELECTEDVALUE('Crime Types Slicer'[Crime Type]),
        "AUTO THEFT",
            CALCULATE(
                COUNTROWS('ALL_CRIMES'),
                SEARCH("MOTOR VEHICLE THEFT", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0,
                VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
                VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
            ),
        "BURGLARY AUTO",
            CALCULATE(
                COUNTROWS('ALL_CRIMES'),
                SEARCH("BURGLARY - AUTO", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0,
                VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
                VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
            ),
        "BURGLARY COMM_RES",
            CALCULATE(
                COUNTROWS('ALL_CRIMES'),
                OR(
                    SEARCH("BURGLARY COMMERCIAL", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0,
                    SEARCH("BURGLARY - RESIDENCE", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0
                ),
                VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
                VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
            ),
        "ROBBERY",
            CALCULATE(
                COUNTROWS('ALL_CRIMES'),
                SEARCH("ROBBERY", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0,
                VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
                VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
            ),
        "SEXUAL OFFENSES",
            CALCULATE(
                COUNTROWS('ALL_CRIMES'),
                SEARCH("SEXUAL", 'ALL_CRIMES'[ALL_INCIDENTS], 1, 0) > 0,
                VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
                VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
            ),
        CALCULATE(
            COUNTROWS('ALL_CRIMES'),
            VALUE('ALL_CRIMES'[IncidentDate]) >= Current7DayStart,
            VALUE('ALL_CRIMES'[IncidentDate]) <= Current7DayEnd
        )
    )

RETURN
IF(
    Count7Day = 0,
    "0 " & CrimeType & " during 7-Day Period (" & StartDateText & "-" & EndDateText & ")",
    Count7Day & " " & CrimeType & " during 7-Day Period (" & StartDateText & "-" & EndDateText & ")"
)

21. DynamicSubtitle =
VAR SelectedCategory =
    SELECTEDVALUE( ALL_CRIMES[Crime_Category], "All Categories" )

VAR TotalCount =
    CALCULATE(
        DISTINCTCOUNT( ALL_CRIMES[Case Number] ),
        REMOVEFILTERS( ALL_CRIMES[Incident Date] )
    )

VAR YTDCount =
    CALCULATE(
        DISTINCTCOUNT( ALL_CRIMES[Case Number] ),
        REMOVEFILTERS( ALL_CRIMES[Incident Date] ),
        FILTER(
            ALL( ALL_CRIMES[Incident Date] ),
            YEAR( ALL_CRIMES[Incident Date] ) = YEAR( TODAY() )
        )
    )

VAR Last28Count =
    CALCULATE(
        DISTINCTCOUNT( ALL_CRIMES[Case Number] ),
        REMOVEFILTERS( ALL_CRIMES[Incident Date] ),
        FILTER(
            ALL( ALL_CRIMES[Incident Date] ),
            ALL_CRIMES[Incident Date] >= TODAY() - 28
                && ALL_CRIMES[Incident Date] <= TODAY()
        )
    )

VAR Last7Count =
    CALCULATE(
        DISTINCTCOUNT( ALL_CRIMES[Case Number] ),
        REMOVEFILTERS( ALL_CRIMES[Incident Date] ),
        FILTER(
            ALL( ALL_CRIMES[Incident Date] ),
            ALL_CRIMES[Incident Date] >= TODAY() - 7
                && ALL_CRIMES[Incident Date] <= TODAY()
        )
    )

RETURN
    SelectedCategory
        & " | Total: "  & FORMAT( TotalCount,  "#,0" )
        & " | YTD: "    & FORMAT( YTDCount,    "#,0" )
        & " | 28-Day: " & FORMAT( Last28Count,"#,0" )
        & " | 7-Day: "  & FORMAT( Last7Count,  "#,0" )
