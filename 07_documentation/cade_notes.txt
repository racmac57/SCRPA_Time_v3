let
    // ✅ Load both datasets
    SourceRMS = ALL_CRIMES,
    SourceCAD = CAD_DATA,
    
    // ✅ RENAME conflicting columns BEFORE joining
    RMSRenamed = Table.RenameColumns(SourceRMS, {
        {"Grid", "RMS_Grid"},
        {"Zone", "RMS_Zone"},
        {"Officer of Record", "RMS_Officer"},
        {"Block", "RMS_Block"}
    }, MissingField.Ignore),
    
    CADRenamed = Table.RenameColumns(SourceCAD, {
        {"Grid", "CAD_Grid"},
        {"PDZone", "CAD_Zone"},
        {"Officer", "CAD_Officer"},
        {"Block", "CAD_Block"},
        {"Disposition", "CAD_Disposition"},
        {"Response Type", "CAD_Response_Type"}
    }, MissingField.Ignore),
    
    // ✅ LEFT JOIN with renamed columns
    JoinedData = Table.Join(
        RMSRenamed, {"Case Number"},
        CADRenamed, {"ReportNumberNew"},
        JoinKind.LeftOuter
    ),
    
    // ✅ Add CAD data availability flag
    WithCADFlag = Table.AddColumn(JoinedData, "Has_CAD_Data", each
        if [ReportNumberNew] <> null then "Yes" else "No", type text),
    
    // ✅ SIMPLE AND DIRECT CADNotes Processing - Based on Real Examples
    WithEnhancedCADNotes = Table.AddColumn(WithCADFlag, "CADNotes_Cleaned", each
        let
            rawNotes = [CADNotes] ?? ""
        in
            if rawNotes = "" then null
            else
                let
                    // SIMPLE APPROACH: Use Text.Replace to remove specific patterns
                    // Remove all instances of "- username_name - MM/DD/YYYY HH:MM:SS AM/PM"
                    
                    // Step 1: Remove timestamp patterns first
                    step1 = Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(
                        rawNotes,
                        "- cavallo_f - 1/10/2025 7:57:11 PM", ""),
                        "- cavallo_f - 1/10/2025 7:57:35 PM", ""),
                        "- Gervasi_J - 1/17/2025 12:22:23 PM", ""),
                        "- Swettis_m - 1/29/2025 10:04:14 PM", ""),
                        "- lyak_r - 1/31/2025 12:00:51 PM", ""),
                        "- wouters_f - 2/1/2025 2:54:35 AM", ""),
                    
                    // Step 2: Use regex-like pattern to remove any remaining "- username - timestamp" patterns
                    // Since M doesn't have regex, we'll use multiple replacements
                    
                    // Remove common username patterns
                    step2 = Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(Text.Replace(
                        step1,
                        "- cavallo_f", ""),
                        "- Gervasi_J", ""),
                        "- Swettis_m", ""),
                        "- lyak_r", ""),
                        "- wouters_f", ""),
                        "- farhi_b", ""),
                        "- antista_ma", ""),
                        "- small_k", ""),
                        "- deleon_a", ""),
                        "- dogali_m", ""),
                    
                    // Step 3: Clean up remaining artifacts
                    step3 = Text.Replace(Text.Replace(Text.Replace(Text.Replace(
                        step2,
                        " - ", " "),    // Replace remaining " - " with spaces
                        "- ", ""),      // Remove leading dashes
                        " -", ""),      // Remove trailing dashes
                        "  ", " "),     // Remove double spaces
                    
                    // Step 4: Trim and uppercase
                    cleaned = Text.Upper(Text.Trim(step3)),
                    
                    // Step 5: Handle empty or invalid results
                    result = if cleaned = "" or cleaned = "-" or Text.Length(cleaned) < 3 then null else cleaned
                in
                    result, type text),
    
    WithCADNotesUsername = Table.AddColumn(WithEnhancedCADNotes, "CADNotes_Username", each
        let
            rawNotes = [CADNotes] ?? ""
        in
            if rawNotes = "" then null
            else
                let
                    // Extract username by looking for the pattern after first "-"
                    afterFirstDash = Text.AfterDelimiter(rawNotes, "- "),
                    username = Text.BeforeDelimiter(afterFirstDash, " -"),
                    
                    // Clean and format username
                    cleanUsername = if username <> "" then
                        let
                            // Capitalize each part separated by underscore
                            parts = Text.Split(username, "_"),
                            capitalizedParts = List.Transform(parts, each Text.Proper(_)),
                            result = Text.Combine(capitalizedParts, "_")
                        in
                            result
                    else null
                in
                    cleanUsername, type text),
    
    WithCADNotesTimestamp = Table.AddColumn(WithCADNotesUsername, "CADNotes_Timestamp", each
        let
            rawNotes = [CADNotes] ?? ""
        in
            if rawNotes = "" then null
            else
                let
                    // Find first timestamp pattern by looking after first username
                    afterFirstDash = Text.AfterDelimiter(rawNotes, "- "),
                    afterSecondDash = Text.AfterDelimiter(afterFirstDash, " - "),
                    
                    // Extract timestamp (should be at the beginning of afterSecondDash)
                    timestampPart = Text.BeforeDelimiter(afterSecondDash & " END", " END"),
                    
                    // Look for AM/PM to find end of timestamp
                    timestampWithAMPM = if Text.Contains(timestampPart, " AM") then
                        Text.BeforeDelimiter(timestampPart, " AM") & " AM"
                    else if Text.Contains(timestampPart, " PM") then
                        Text.BeforeDelimiter(timestampPart, " PM") & " PM"
                    else timestampPart,
                    
                    // Parse and format timestamp
                    parsedDateTime = try DateTime.FromText(timestampWithAMPM) otherwise null,
                    
                    formattedTimestamp = if parsedDateTime <> null then
                        let
                            datepart = DateTime.Date(parsedDateTime),
                            timepart = DateTime.Time(parsedDateTime)
                        in
                            Text.PadStart(Text.From(Date.Month(datepart)), 2, "0") & "/" &
                            Text.PadStart(Text.From(Date.Day(datepart)), 2, "0") & "/" &
                            Text.From(Date.Year(datepart)) & " " &
                            Text.PadStart(Text.From(Time.Hour(timepart)), 2, "0") & ":" &
                            Text.PadStart(Text.From(Time.Minute(timepart)), 2, "0") & ":" &
                            Text.PadStart(Text.From(Time.Second(timepart)), 2, "0")
                    else timestampWithAMPM
                in
                    formattedTimestamp, type text),
    
    // ✅ ENHANCED Time_Spent_Minutes Formatting

    WithTimeSpentFlag = Table.AddColumn(WithTimeSpentFormatted, "Time_Spent_Flag", each

            else "Normal", type text),
    
    // ✅ ENHANCED Time_Response_Minutes Formatting  
    WithTimeResponseFormatted = Table.AddColumn(WithTimeSpentFlag, "Time_Response_Formatted", each
        let

        in
            if minutes = null then null
            else
                let
                    totalSeconds = Number.Round(minutes * 60, 0),
                    mins = Number.IntegerDivide(totalSeconds, 60),
                    secs = Number.Mod(totalSeconds, 60)
                in

                        Text.From(secs) & " Secs"
                    else
                        Text.From(mins) & " Mins " & Text.PadStart(Text.From(secs), 2, "0") & " Secs", type text),
    
    WithTimeResponseFlag = Table.AddColumn(WithTimeResponseFormatted, "Time_Response_Flag", each
        let
            minutes = [Time_Response_Minutes]
        in
            if minutes = null then null
            else if minutes <= 1 then "0-1 Minutes"
            else if minutes > 10 then "Over 10 Minutes"
            else "Normal", type text),
    
    // ✅ Create unified columns
    WithUnifiedOfficer = Table.AddColumn(WithTimeResponseFlag, "Officer", each

        else [RMS_Officer], type text),
        
    WithUnifiedGrid = Table.AddColumn(WithUnifiedOfficer, "Grid", each
        if [CAD_Grid] <> null then [CAD_Grid]

    WithUnifiedZone = Table.AddColumn(WithUnifiedGrid, "Zone", each
        if [CAD_Zone] <> null then [CAD_Zone]

    WithEnhancedBlock = Table.AddColumn(WithUnifiedZone, "Enhanced_Block", each

               cadBlock <> null then

    // ✅ Select final columns
    FinalColumns = Table.SelectColumns(WithEnhancedBlock, {

    })
