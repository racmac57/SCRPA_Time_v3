def parse_date_string(date_str):
    """
    Parse date string with support for special keywords
    
    Args:
        date_str (str): Date string in YYYY_MM_DD format or special keyword
        
    Returns:
        date: Parsed date object
    """
    from datetime import date, timedelta
    import re
    
    if not date_str:
        return date.today()
    
    # Handle special keywords
    if date_str.lower() == 'today':
        return date.today()
    elif date_str.lower() == 'yesterday':
        return date.today() - timedelta(days=1)
    elif date_str.lower() == 'tomorrow':
        return date.today() + timedelta(days=1)

        # Try YYYY-MM-DD format (with dashes)
        try:
            return dt_datetime.strptime(date_str, "%Y-%m-%d").date()
        except ValueError:
            # Try MM/DD/YYYY format
            try:
                return dt_datetime.strptime(date_str, "%m/%d/%Y").date()
            except ValueError:
                raise ValueError(f"Invalid date format: {date_str}. Use YYYY_MM_DD, 'today', 'yesterday', or 'tomorrow'")
