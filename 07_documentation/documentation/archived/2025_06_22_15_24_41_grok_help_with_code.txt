def extract_block_from_address(address_text):
    """
    Extract block information from FullAddress2 field using M Code logic (CODE_20250622_001).
    Converts address like "327 Main St" to "300 BLOCK"
    """
    try:
        if not address_text or address_text.strip() == "":
            return "Unknown Block"
        address = str(address_text).upper().strip()
        parts = address.split(" ")
        if len(parts) == 0:
            return "Unknown Block"
        first_word = parts[0]
        try:
            block_number = int(first_word)
            if block_number > 0:
                block_hundred = (block_number // 100) * 100
                return f"{block_hundred} BLOCK"
            else:
                return "Unknown Block"
        except ValueError:
            return f"BLOCK {address[:30]}"
    except Exception as e:
        logging.error(f"⚠️ Block extraction error: {e}")
        return "Unknown Block"
def find_field(available_fields, candidates, field_type):
    """Helper function to find field by trying multiple candidate names"""
    for candidate in candidates:
        if candidate in available_fields:
            logging.info(f"✅ Found {field_type} field: {candidate}")
            return candidate
    logging.warning(f"❌ No {field_type} field found in candidates: {candidates}")
    return None
def get_7day_incident_details(crime_type, target_date):
    """Extract detailed incident information with proper block extraction"""
    incidents = []
    try:
        start_date, end_date = get_7day_period_dates(target_date)
        if not start_date or not end_date:
            logging.error(f"❌ Could not determine Excel 7-day period for {target_date}")
            return incidents
        logging.info(f"📅 Getting incident details for period: {start_date} to {end_date}")
        aprx = arcpy.mp.ArcGISProject(APR_PATH)
        maps = aprx.listMaps()
        if not maps:
            logging.error(f"❌ No maps found in project")
            return incidents
        map_obj = maps[0]
        layer_name = f"{crime_type} 7-Day"
        target_layer = None
        for lyr in map_obj.listLayers():
            if lyr.name == layer_name:
                target_layer = lyr
                break
        if not target_layer:
            logging.error(f"❌ 7-Day layer not found: {layer_name}")
            return incidents
        logging.info(f"🔍 FIELD ANALYSIS for {layer_name}:")
        try:
            available_fields = [field.name for field in arcpy.ListFields(target_layer)]
            logging.info(f"📋 Available fields: {available_fields}")
        except Exception as e:
            logging.error(f"⚠️ Field inspection error: {e}")
            available_fields = ['OBJECTID', 'calldate', 'calltype', 'disposition']
        field_mapping = {}
        date_candidates = ['calldate', 'call_date', 'date', 'incident_date', 'Date', 'CallDate', 'CALLDATE']
        field_mapping['date'] = find_field(available_fields, date_candidates, 'date')
        address_candidates = [
            'FullAddress2', 'fulladdress2', 'FULLADDRESS2',
            'FullAddress', 'fulladdress', 'FULLADDRESS',
            'address', 'Address', 'ADDRESS',
            'location', 'Location', 'LOCATION',
            'block', 'Block', 'BLOCK',
            'street', 'Street', 'STREET'
        ]
        field_mapping['location'] = find_field(available_fields, address_candidates, 'address/location')
        calltype_candidates = ['calltype', 'call_type', 'type', 'incident_type', 'CallType', 'CALLTYPE']
        field_mapping['calltype'] = find_field(available_fields, calltype_candidates, 'calltype')
        disposition_candidates = ['disposition', 'disp', 'status', 'Disposition', 'DISPOSITION', 'Status']
        field_mapping['disposition'] = find_field(available_fields, disposition_candidates, 'disposition')
        logging.info(f"📋 FINAL FIELD MAPPING: {field_mapping}")
        if not field_mapping['location']:
            logging.warning("⚠️ No address field found, using geometry")
            field_mapping['location'] = 'SHAPE@XY'
        from map_export import build_sql_filter_7day_excel
        filter_sql = build_sql_filter_7day_excel(crime_type, start_date, end_date)
        if filter_sql:
            original_query = target_layer.definitionQuery
            target_layer.definitionQuery = filter_sql
            try:
                fields_to_extract = [
                    field_mapping['date'],
                    field_mapping['location'], 
                    field_mapping['calltype'],
                    field_mapping['disposition']
                ]
                logging.info(f"🔍 Extracting with fields: {fields_to_extract}")
                with arcpy.da.SearchCursor(target_layer, fields_to_extract) as cursor:
                    for row in cursor:
                        try:
                            calldate = row[0]
                            location_data = row[1]
                            calltype = row[2] if row[2] else "Unknown"
                            disposition = row[3] if row[3] else "Unknown"
                            if field_mapping['location'] == 'SHAPE@XY':
                                if location_data and len(location_data) >= 2:
                                    x, y = location_data[0], location_data[1]
                                    block_str = f"BLOCK {abs(int(x/1000))}{abs(int(y/1000))}"
                                else:
                                    block_str = "Unknown Block"
                            else:
                                block_str = extract_block_from_address(location_data)
                            if isinstance(calldate, datetime):
                                hour = calldate.hour
                                if 0 <= hour <= 3:
                                    time_period = "Early Morning"
                                    time_range = "00:00-03:59"
                                elif 4 <= hour <= 7:
                                    time_period = "Morning"
                                    time_range = "04:00-07:59"
                                elif 8 <= hour <= 11:
                                    time_period = "Midday"
                                    time_range = "08:00-11:59"
                                elif 12 <= hour <= 15:
                                    time_period = "Afternoon"
                                    time_range = "12:00-15:59"
                                elif 16 <= hour <= 19:
                                    time_period = "Evening"
                                    time_range = "16:00-19:59"
                                elif 20 <= hour <= 23:
                                    time_period = "Night"
                                    time_range = "20:00-23:59"
                                else:
                                    time_period = "Unknown"
                                    time_range = "Unknown"
                                incident = {
                                    'block': block_str,
                                    'incident_date': calldate.strftime('%m/%d/%y'),
                                    'day_of_week': calldate.strftime('%A'),
                                    'time_period': time_period,
                                    'time_of_day': time_range,
                                    'raw_datetime': calldate,
                                    'vehicle_description': '[MANUAL ENTRY]',
                                    'suspect_description': '[MANUAL ENTRY]',
                                    'narrative_summary': '[MANUAL ENTRY]'
                                }
                                incidents.append(incident)
                                logging.info(f"✅ Extracted: {calldate.strftime('%m/%d/%y %H:%M')} at {block_str}")
                        except Exception as e:
                            logging.error(f"⚠️ Error processing record: {e}")
                            continue
                target_layer.definitionQuery = original_query
                incidents.sort(key=lambda x: x['raw_datetime'])
                logging.info(f"✅ Successfully extracted {len(incidents)} incidents for {crime_type}")
            except Exception as e:
                logging.error(f"❌ Error during extraction: {e}")
                try:
                    target_layer.definitionQuery = original_query
                except:
                    pass
        del aprx
    except Exception as e:
        logging.error(f"❌ Critical error in incident extraction: {e}")
    return incidents
def create_formatted_csv_image_with_autofit(csv_path, output_image_path):
    """
    Create CSV image with auto-fitted column widths, bold headers, center alignment
    """
    if not pd:
        logging.error("❌ Pandas not available, cannot create auto-fit table")
        return False
    try:
        df = pd.read_csv(csv_path)
        if df.empty:
            logging.warning(f"⚠️ CSV file is empty: {csv_path}")
            return False
        logging.info(f"📊 Creating auto-fit table image for {len(df)} rows, {len(df.columns)} columns")
        column_widths = []
        min_width = 0.8
        max_width = 3.0
        for col in df.columns:
            header_length = len(str(col))
            data_lengths = [len(str(val)) for val in df[col] if pd.notna(val)]
            max_data_length = max(data_lengths) if data_lengths else 0
            max_length = max(header_length, max_data_length)
            estimated_width = max_length * 0.08 + 0.3
            column_width = max(min_width, min(estimated_width, max_width))
            column_widths.append(column_width)
            logging.info(f"  📏 Column '{col}': max_chars={max_length}, width={column_width:.2f}in")
        total_width = sum(column_widths) + 1
        figure_height = max(3, len(df) * 0.3 + 1)
        logging.info(f"📐 Figure dimensions: {total_width:.1f}\" x {figure_height:.1f}\"")
        fig, ax = plt.subplots(figsize=(total_width, figure_height))
        ax.axis('tight')
        ax.axis('off')
        table_data = []
        table_data.append(df.columns.tolist())
        table_data.extend(df.values.tolist())
        table = ax.table(cellText=table_data, 
                        loc='center',
                        cellLoc='center')
        total_table_width = 1.0
        for col_idx, width in enumerate(column_widths):
            proportional_width = width / total_width
            for row_idx in range(len(table_data)):
                cell = table[(row_idx, col_idx)]
                cell.set_width(proportional_width)
                cell.set_height(0.08 if row_idx == 0 else 0.06)
        table.auto_set_font_size(False)
        table.set_fontsize(9)
        for col in range(len(df.columns)):
            cell = table[(0, col)]
            cell.set_text_props(weight='bold', family='Arial', ha='center')
            cell.set_facecolor('#E6E6FA')
            cell.set_edgecolor('#000080')
            cell.set_linewidth(2)
            cell.get_text().set_fontweight('bold')
        for row in range(1, len(table_data)):
            for col in range(len(df.columns)):
                cell = table[(row, col)]
                cell.set_text_props(family='Arial', ha='center')
                cell.set_facecolor('white' if row % 2 == 1 else '#F8F8FF')
                cell.set_edgecolor('#DDDDDD')
                cell.set_linewidth(0.5)
        plt.subplots_adjust(left=0.05, right=0.95, top=0.95, bottom=0.05)
        plt.savefig(output_image_path, dpi=200, bbox_inches='tight', 
                   facecolor='white', edgecolor='none', pad_inches=0.2)
        plt.close()
        logging.info(f"✅ Created auto-fit CSV image: {output_image_path}")
        return True
    except Exception as e:
        logging.error(f"❌ Error creating auto-fit CSV image: {e}")
        return False

        header_height = 30
        draw.rectangle([20, header_y, width-20, header_y + header_height],

        for col in columns:
            text_x = col['start_x'] + 5
            if col['align'] == 'center':
                text_bbox = draw.textbbox((0, 0), col['name'], font=header_font)
                text_width = text_bbox[2] - text_bbox[0]

                incident.get('narrative_summary', '')[:55] + ('...' if len(incident.get('narrative_summary', '')) > 55 else '')
